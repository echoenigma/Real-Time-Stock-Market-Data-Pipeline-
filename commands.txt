7   Trend Detection (Short-Term) 
    SELECT 
    time,
    close,
    LAG(close, 5) OVER (ORDER BY time) AS close_5min_ago,
    (close - LAG(close, 5) OVER (ORDER BY time)) AS delta
FROM ohlc
ORDER BY time DESC
LIMIT 1;
